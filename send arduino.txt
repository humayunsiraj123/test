#include <Arduino.h>
#include <SoftwareSerial.h>

SoftwareSerial comm(5,6);


void setup() {
  comm.begin(9600);
}

void loop() {
  comm.println("From Arduino"); // send to ESP

  char buffer[20] = "";
  if (Serial.available())
  {
    Serial.readBytesUntil('\n', buffer, 20);
    Serial.println(buffer); // recieve from ESP
  }

  delay(1000);
}
